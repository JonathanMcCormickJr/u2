13// Single enrolment or pairing between an application and a token.

25pub fn parse_registration(
26    app_id: String,
27    client_data: Vec<u8>,
28    registration_data: Vec<u8>,
29) -> Result<Registration> {
30    let reserved_byte = registration_data[0];
31    if reserved_byte != 0x05 {
32        return Err(U2fError::InvalidReservedByte);
33    }
34
35    let mut mem = Bytes::from(registration_data);
36
37    //Start parsing ... advance the reserved byte.
38    let _ = mem.split_to(1);
39
40    // P-256 NIST elliptic curve
41    let public_key = mem.split_to(65);
42
43    // Key Handle
44    let key_handle_size = mem.split_to(1);
45    let key_len = BigEndian::read_uint(&key_handle_size[..], 1);
46    let key_handle = mem.split_to(key_len as usize);
47
48    // The certificate length needs to be inferred by parsing.
49    let cert_len = asn_length(mem.clone()).unwrap();
50    let attestation_certificate = mem.split_to(cert_len);
51
52    // Remaining data corresponds to the signature
53    let signature = mem;
54
55    // Let's build the msg to verify the signature
56    let app_id_hash = sha256(&app_id.into_bytes());
57    let client_data_hash = sha256(&client_data[..]);
58
59    let mut msg = vec![0x00]; // A byte reserved for future use [1 byte] with the value 0x00
60    msg.put(app_id_hash.as_ref());
61    msg.put(client_data_hash.as_ref());
62    msg.put(key_handle.clone());
63    msg.put(public_key.clone());
64
65    // The signature is to be verified by the relying party using the public key certified
66    // in the attestation certificate.
67    let cerificate_public_key =
68        super::crypto::X509PublicKey::try_from(&attestation_certificate[..])?;
69
70    if !(cerificate_public_key.is_secp256r1()?) {
71        return Err(U2fError::BadCertificate);
72    }
73
74    let verified = cerificate_public_key.verify_signature(&signature[..], &msg[..])?;
75
76    if !verified {
77        return Err(U2fError::BadCertificate);
78    }
79
80    let registration = Registration {
81        key_handle: key_handle[..].to_vec(),
82        pub_key: public_key[..].to_vec(),
83        attestation_cert: Some(attestation_certificate[..].to_vec()),
84        device_name: cerificate_public_key.common_name(),
85    };
86
87    Ok(registration)
88}
